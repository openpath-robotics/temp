154  if (NormalizeTexture) { \
155  if (Source.Texture.Size.Area() > 0) { \
156  const double SamplesCount = 10000.0; \
157  const double SizeRatio = (double)Source.Texture.Size.Width / (double)Source.Texture.Size.Height; \
158  int SamplesX = (int)std::sqrt(SamplesCount * SizeRatio); \
159  int SamplesY = (int)std::sqrt(SamplesCount / SizeRatio); \
160  std::vector<float> Samples; \
161  int StepY = std::max<int>(1, Source.Texture.Size.Height / SamplesY); \
162  int StepX = std::max<int>(1, Source.Texture.Size.Width / SamplesX); \
163  for (int y = 0; y < Source.Texture.Size.Height; y += StepY) { \
164  for (int x = 0; x < Source.Texture.Size.Height; x += StepY) { \
165  Samples.push_back(Source.Texture[y][x]); \
166  } \
167  } \
168  if (Samples.size() > 10) { \
169  std::sort(Samples.begin(), Samples.end()); \
170  TextureMultiplier = (255.0 / Samples[Samples.size() - 1 - Samples.size() / 98]) * 0.9; \
171  } \
172  } else if (Source.TextureRGB.Size.Area() > 0) { \
173  const double SamplesCount = 10000.0; \
174  const double SizeRatio = (double)Source.TextureRGB.Size.Width / (double)Source.TextureRGB.Size.Height; \
175  int SamplesX = (int)std::sqrt(SamplesCount * SizeRatio); \
176  int SamplesY = (int)std::sqrt(SamplesCount / SizeRatio); \
177  using CombinedColorT = std::pair<ColorRGB_16, float>; \
178  std::vector<CombinedColorT> Samples; \
179  int StepY = std::max<int>(1, Source.TextureRGB.Size.Height / SamplesY); \
180  int StepX = std::max<int>(1, Source.TextureRGB.Size.Width / SamplesX); \
181  for (int y = 0; y < Source.TextureRGB.Size.Height; y += StepY) { \
182  for (int x = 0; x < Source.TextureRGB.Size.Height; x += StepY) { \
183  const ColorRGB_16& rgb = Source.TextureRGB[y][x]; \
184  Samples.emplace_back(std::make_pair(rgb, 0.11f * rgb.r + 0.59f * rgb.g + 0.3f * rgb.b)); \

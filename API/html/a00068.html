template<typename T, class Enable = void>
struct pho::api::SharedPtrCreator< T, Enable >

SharedPtrCreator is a helper class for creating shared_ptr object from any instance. Class detects if instance is derived from enable_shared_from_this and then create shared_ptr from this.

385  Result *= SizeND[i];

394  if ((int) SizeND.size() != Dimension) return false;

396  if (ElementsCountNew != ElementsCount) return false;

407 
408  if (ElementsCountNew != ElementsCount) {
409  Data = std::unique_ptr < ElementType, std::function< void(ElementType*) >> (new ElementType[ElementsCountNew], [](ElementType* Ptr) { delete[] Ptr; });
